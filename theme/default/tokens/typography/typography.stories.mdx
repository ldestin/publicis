import { Grid, Typography } from '@material-ui/core';
import { ThemeProvider } from '@spark/themes';
import { theme } from '@spark/themes/default';
const { typography } = theme;
export const text = 'The quick brown fox jumps over the lazy dog';
const notMixin = [
  'htmlFontSize',
  'fontFamily',
  'fontSize',
  'fontWeightLight',
  'fontWeightRegular',
  'fontWeightMedium',
  'fontWeightBold',
  'fontWeightBlack',
  'fontSizes',
  'lineHeights',
  'letterSpacings',
];

export const fontWeights = {
  light: typography.fontWeightLight,
  regular: typography.fontWeightRegular,
  medium: typography.fontWeightMedium,
  bold: typography.fontWeightBold,
  black: typography.fontWeightBlack,
};
